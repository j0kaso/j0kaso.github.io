def num_particles(concentration, volume):
    """Given a molar concentration in mol/L and volume
    in cubic angstroms, returns the number of particles"""
    avogadro_num = 6.022E23
    angstrom_to_liters = 1E-27
    particles_per_liter = concentration * avogadro_num
    liters = volume * angstrom_to_liters

    return particles_per_liter * liters

# example usage

import matplotlib.pyplot as plt
import matplotlib.mlab as mlab
from scipy.interpolate import spline
import numpy as np

%pylab inline


def errorbarplotting(x, y, std_y, title="", xlabel="", ylabel="", everyerror=0, x_ticks=[]):
    
    fig, ax = plt.subplots(figsize=(10, 5))
    plt.tick_params(labelsize=20)
    
    x_sm = np.array(x)
    y_sm = np.array(y)
    
    for index, elem in enumerate(std_y):
        if elem < 0:
            std_y[index] = elem*-1
    std_sm = np.array(std_y)
    
    if everyerror !=0:
        plt.errorbar(x_sm, y_sm, yerr=std_sm, errorevery=everyerror)
    else:
        plt.errorbar(x_sm, y_sm, yerr=std_sm)
    plt.title(title, fontsize=20)
    plt.xlabel(xlabel, fontsize=20)
    plt.ylabel(ylabel, fontsize=20)
    if x_ticks != []:
        plt.xticks(np.arange(len(x_ticks)), x_ticks, fontsize=20)
    plt.show()


def smooth_errorareaplotting(x, y, std_y, std_y2=[], title="", xlabel="", ylabel="", x_ticks=[]):
    
    fig, ax = plt.subplots(figsize=(10, 5))
    plt.tick_params(labelsize=20)
    
    x_sm = np.array(x)
    y_sm = np.array(y)
    std_sm = np.array(std_y)
    
    x_smooth = np.linspace(x_sm.min(), x_sm.max(), 200)
    y_smooth = spline(x_sm, y_sm, x_smooth)

    if std_y2 == []:
        yu = np.zeros(len(std_sm))
        yd = np.zeros(len(std_sm))
        for index, elem in enumerate(std_sm):
            if elem >= 0:
                yu[index] = y_sm[index]-elem
                yd[index] = y_sm[index]+elem
            else:
                yd[index] = y_sm[index]-elem
                yu[index] = y_sm[index]+elem
    else:
        yu = np.array(std_sm)
        std2_sm = np.array(std_y2)
        yd = np.array(std2_sm)

    yu_smooth = spline(x_sm, yu, x_smooth)
    yd_smooth = spline(x_sm, yd, x_smooth)

    plt.plot(x_smooth, y_smooth, 'black', linewidth=1)

    plt.fill_between(x_smooth, yu_smooth, yd_smooth, facecolor='lightgrey', alpha=0.5, interpolate=True)
    plt.grid(True)
    plt.title(title, fontsize=20)
    plt.xlabel(xlabel, fontsize=20)
    plt.ylabel(ylabel, fontsize=20)
    if x_ticks != []:
        plt.xticks(np.arange(len(x_ticks)), x_ticks, fontsize=20)
    plt.show()

def autolabel(rects, xpos='center'):
    """
    Attach a text label above each bar in *rects*, displaying its height.

    *xpos* indicates which side to place the text w.r.t. the center of
    the bar. It can be one of the following {'center', 'right', 'left'}.
    """

    xpos = xpos.lower()  # normalize the case of the parameter
    ha = {'center': 'center', 'right': 'left', 'left': 'right'}
    offset = {'center': 0.5, 'right': 0.57, 'left': 0.43}  # x_txt = x + w*off

    for rect in rects:
        height = rect.get_height()
        ax.text(rect.get_x() + rect.get_width()*offset[xpos], 1.01*height,
                '{}'.format(height), ha=ha[xpos], va='bottom')


autolabel(rects1, "left")
autolabel(rects2, "right")

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.ticker import NullFormatter

# Fixing random state for reproducibility
np.random.seed(19680801)


# the random data
x = np.random.randn(1000)
y = np.random.randn(1000)

nullfmt = NullFormatter()         # no labels

# definitions for the axes
left, width = 0.1, 0.65
bottom, height = 0.1, 0.65
bottom_h = left_h = left + width + 0.02

rect_scatter = [left, bottom, width, height]
rect_histx = [left, bottom_h, width, 0.2]
rect_histy = [left_h, bottom, 0.2, height]

# start with a rectangular Figure
plt.figure(1, figsize=(8, 8))

axScatter = plt.axes(rect_scatter)
axHistx = plt.axes(rect_histx)
axHisty = plt.axes(rect_histy)

# no labels
axHistx.xaxis.set_major_formatter(nullfmt)
axHisty.yaxis.set_major_formatter(nullfmt)

# the scatter plot:
axScatter.scatter(x, y)

# now determine nice limits by hand:
binwidth = 0.25
xymax = max(np.max(np.abs(x)), np.max(np.abs(y)))
lim = (int(xymax/binwidth) + 1) * binwidth

axScatter.set_xlim((-lim, lim))
axScatter.set_ylim((-lim, lim))

bins = np.arange(-lim, lim + binwidth, binwidth)
axHistx.hist(x, bins=bins)
axHisty.hist(y, bins=bins, orientation='horizontal')

axHistx.set_xlim(axScatter.get_xlim())
axHisty.set_ylim(axScatter.get_ylim())

plt.show()
